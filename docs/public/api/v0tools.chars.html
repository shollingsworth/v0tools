def unicodes():

yield all Unicode values.

returns iterator of dictionaries in the following format:

    {
        "int": i,
        "hex": hval,
        "chr": char,
        "name": name.lower(),
        "pref": pref,
        "htmlent": htmlent
    }

example for letter "a":

    {
        "int": 97,
        "hex": "61",
        "chr": "a",
        "name": "latin small letter a",
        "pref": "\\u0061",
        "htmlent": "&#97;"
    }
